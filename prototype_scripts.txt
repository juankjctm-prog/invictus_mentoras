        function iniciarDiagLectura() {
            diagLecturaStart = Date.now();
            document.getElementById('diag-btn-iniciar').style.display = 'none';
            document.getElementById('diag-btn-terminar').style.display = 'block';
            diagTimerInterval = setInterval(() => {
                const s = Math.floor((Date.now() - diagLecturaStart) / 1000);
                const m = Math.floor(s/60).toString().padStart(2,'0');
                const ss = (s%60).toString().padStart(2,'0');
                document.getElementById('diag-lectura-timer').textContent = m+':'+ss;
            }, 1000);
        }

        function terminarDiagLectura() {
            clearInterval(diagTimerInterval);
            diagLecturaElapsed = Math.max(0.1, (Date.now() - diagLecturaStart) / 60000);
            renderDiagStep(2);
        }

        function renderQuestions(questions, type, nextStep) {
            let html = '';
            questions.forEach((q, idx) => {
                html += '<div class="diag-question" data-idx="'+idx+'" data-type="'+type+'"><p><strong>'+(idx+1)+'.</strong> '+q.q+'</p>';
                q.options.forEach((opt, oidx) => {
                    html += '<div class="diag-option" onclick="selectOption(this, \''+type+'\', '+idx+', '+oidx+')">' +
                        '<input type="radio" name="'+type+'-q'+idx+'" value="'+oidx+'">' +
                        '<span>'+opt+'</span></div>';
                });
                html += '</div>';
            });
            const answered = diagAnswers[type].filter(a => a !== undefined).length;
            html += '<button id="diag-btn-next" class="diag-btn" onclick="validateAndGo('+nextStep+', \''+type+'\')" ' + (answered === questions.length ? '' : 'disabled') + '>Continuar →</button>';
            html += '<p style="text-align:center;font-size:0.75rem;color:var(--text-tertiary);margin-top:10px;">Respondidas: <span id="diag-count">'+answered+'</span> / '+questions.length+'</p>';
            return html;
        }

        function selectOption(el, type, qidx, oidx) {
            const qDiv = el.parentElement;
            qDiv.querySelectorAll('.diag-option').forEach(o => o.classList.remove('selected'));
            el.classList.add('selected');
            el.querySelector('input').checked = true;
            diagAnswers[type][qidx] = oidx;
            const total = (type === 'lectura' ? DIAG_LECTURA_QS : (type === 'mentoria' ? DIAG_MENTOR_QS : DIAG_BRECHAS_QS)).length;
            const answered = diagAnswers[type].filter(a => a !== undefined).length;
            const btn = document.getElementById('diag-btn-next');
            const count = document.getElementById('diag-count');
            if (btn) btn.disabled = answered < total;
            if (count) count.textContent = answered;
        }

        function validateAndGo(nextStep, type) {
            const total = (type === 'lectura' ? DIAG_LECTURA_QS : (type === 'mentoria' ? DIAG_MENTOR_QS : DIAG_BRECHAS_QS)).length;
            if (diagAnswers[type].filter(a => a !== undefined).length < total) return;
            if (type === 'brechas') { calcularYGuardarResultados(); }
            else { renderDiagStep(nextStep); }
        }

        function calcularYGuardarResultados() {
            const ppm = Math.round(DIAG_LECTURA_WORDS / diagLecturaElapsed);
            let correctasLectura = 0;
            DIAG_LECTURA_QS.forEach((q, i) => { if (diagAnswers.lectura[i] === q.correct) correctasLectura++; });
            const compLectura = Math.round((correctasLectura / DIAG_LECTURA_QS.length) * 100);

            const mentorScores = {};
            const dimCount = {};
            DIAG_MENTOR_QS.forEach((q, i) => {
                mentorScores[q.dimension] = (mentorScores[q.dimension] || 0) + (diagAnswers.mentoria[i] === q.correct ? 1 : 0);
                dimCount[q.dimension] = (dimCount[q.dimension] || 0) + 1;
            });
            const mentorPercent = {};
            for (let d in mentorScores) mentorPercent[d] = Math.round((mentorScores[d] / dimCount[d]) * 100);
            const totalMentoria = Math.round((diagAnswers.mentoria.filter((a,i) => a === DIAG_MENTOR_QS[i].correct).length / DIAG_MENTOR_QS.length) * 100);

            const brechasSuma = {};
            const brechasCount = {};
            DIAG_BRECHAS_QS.forEach((q, i) => {
                const val = (diagAnswers.brechas[i] || 0) + 1;
                brechasSuma[q.dimension] = (brechasSuma[q.dimension] || 0) + val;
                brechasCount[q.dimension] = (brechasCount[q.dimension] || 0) + 1;
            });
            const brechasPercent = {};
            DIAG_BRECHAS_DIMS.forEach(d => { brechasPercent[d] = Math.round(((brechasSuma[d] || 0) / (brechasCount[d] * 4)) * 100); });

            const resultados = {
                fecha: new Date().toISOString(),
                lectura: { ppm: Math.min(ppm, 1200), comprension: compLectura },
                mentoria: { porDimension: mentorPercent, total: totalMentoria },
                brechas: brechasPercent
            };
            localStorage.setItem('mm_diag_inicial', JSON.stringify(resultados));
            mostrarResultados(resultados);
        }

        function mostrarResultados(r) {
            const title = document.getElementById('diag-title');
            const subtitle = document.getElementById('diag-subtitle');
            title.textContent = 'Tu Diagnostico Inicial';
            subtitle.textContent = 'Este es tu punto de partida. Al final del programa compararas tu progreso.';
            document.getElementById('diag-progress-bar').style.width = '100%';

            const brechasSorted = Object.entries(r.brechas).sort((a,b) => a[1] - b[1]);
            const brechasCriticas = brechasSorted.filter(x => x[1] <= 40);

            let brechasHtml = '';
            brechasSorted.forEach(([name, val]) => {
                const cls = val <= 40 ? 'alta' : (val <= 70 ? 'media' : 'baja');
                brechasHtml += '<div class="brecha-item '+cls+'"><span class="brecha-name">'+name+'</span><span class="brecha-val">'+val+'%</span></div>';
            });

            const dims = DIAG_BRECHAS_DIMS;
            const vals = dims.map(d => r.brechas[d] || 0);
            const points = dims.map((d, i) => {
                const angle = (Math.PI * 2 * i / dims.length) - Math.PI/2;
                const v = (vals[i] || 0) / 100;
                const x = 100 + 70 * v * Math.cos(angle);
                const y = 100 + 70 * v * Math.sin(angle);
                return x+','+y;
            }).join(' ');
            const axis = dims.map((d, i) => {
                const angle = (Math.PI * 2 * i / dims.length) - Math.PI/2;
                const x = 100 + 85 * Math.cos(angle);
                const y = 100 + 85 * Math.sin(angle);
                const tx = 100 + 95 * Math.cos(angle);
                const ty = 100 + 95 * Math.sin(angle);

var thetaUtt = null;
var thetaPlaying = false;

window.toggleTheta = function(dayId) {
    var btn = document.getElementById('theta-btn-' + dayId);
    if (thetaPlaying) {
        window.speechSynthesis.cancel();
        thetaPlaying = false;
        if (btn) { btn.textContent = 'Iniciar Meditacion Guiada'; btn.style.background = ''; }
        return;
    }
    var scripts = window.THETA_SCRIPTS || {};
    var script = scripts[dayId] || 'Respira profundo. Integra el aprendizaje de hoy.';
    thetaUtt = new SpeechSynthesisUtterance(script);
    thetaUtt.lang = 'es-ES';
    thetaUtt.rate = 0.82;
    thetaUtt.pitch = 1.0;
    thetaUtt.volume = 1.0;
    thetaUtt.onend = function() {
        thetaPlaying = false;
        if (btn) { btn.textContent = 'Iniciar Meditacion Guiada'; btn.style.background = ''; }
    };
    window.speechSynthesis.speak(thetaUtt);
    thetaPlaying = true;
    if (btn) { btn.textContent = 'Detener'; btn.style.background = 'rgba(0,198,255,0.15)'; }
};

// TIMER LOGIC FOR SPEED READING
let lecturaStartTime = null;
let lecturaInterval = null;

function formatTime(secs) {
    let m = Math.floor(secs / 60).toString().padStart(2, '0');
    let s = (secs % 60).toString().padStart(2, '0');
    return m + ":" + s;
}

function iniciarLecturaPPM() {
    lecturaStartTime = Date.now();
    document.getElementById('reader-content-premium').style.display = 'block';
    document.getElementById('btn-iniciar-lectura').style.display = 'none';
    document.getElementById('btn-terminar-lectura').style.display = 'block';
    
    lecturaInterval = setInterval(() => {
        let elapsed = Math.floor((Date.now() - lecturaStartTime) / 1000);
        document.getElementById('ppm-timer-display').innerText = formatTime(elapsed);
    }, 1000);
}

function terminarLecturaPPM() {
    clearInterval(lecturaInterval);
    let elapsedMs = Date.now() - lecturaStartTime;
    let elapsedMinutes = elapsedMs / 60000;
    
    // Estimate word count by extracting text from the reader content
    let contentEl = document.getElementById('reader-content-premium');
    let text = contentEl.innerText || contentEl.textContent;
    let wordCount = text.trim().split(/\s+/).length;
    
    // Prevent division by zero or super fast clicks
    if (elapsedMinutes < 0.1) elapsedMinutes = 0.1;
    
    let ppm = Math.round(wordCount / elapsedMinutes);
    if (ppm > 1200) ppm = 1200; // Cap to realistic max
    
    // Save locally for stats (Kit de Mentoría uses km_ppm)
    let key = 'mm_b' + window.location.pathname.match(/Bloque(\d+)/)[1] + '_d' + currentDayId + '_ppm';
    localStorage.setItem(key, ppm.toString());
    
    // Render Modal
    document.getElementById('ppm-result').innerText = ppm;
    
    let msg = "";
    if (ppm < 150) msg = "Velocidad base. Evita subvocalizar (repetir palabras mentalmente).";
    else if (ppm < 250) msg = "Velocidad promedio. Usa tu dedo o un puntero para guiar tu vista más rápido.";
    else if (ppm < 400) msg = "¡Buena velocidad! Estás por encima del promedio. Sigue ampliando tu visión periférica.";
    else if (ppm < 600) msg = "¡Nivel Ejecutivo! Excelente comprensión y velocidad.";
    else msg = "¡Nivel Élite! Velocidad de lectura biónica alcanzada.";
    
    document.getElementById('ppm-msg').innerText = msg;
    document.getElementById('ppm-modal').style.display = 'flex';
    
    // Hide timer controls to show completion
    document.getElementById('btn-terminar-lectura').style.display = 'none';
    document.getElementById('ppm-timer-display').style.color = 'var(--success)';
}

function evaluarQuizReal(dayId) {
    const data = daysData.find(d => d.id === dayId);
    if (!data || !data.quiz) return;

    let correctas = 0;
    let sinResponder = 0;
    data.quiz.forEach((item, qIdx) => {
        const radios = document.getElementsByName('quiz-q' + qIdx);
        let answered = false;
        for (let r of radios) {
            if (r.checked) {
                answered = true;
                if (parseInt(r.value) === item.correct) correctas++;
            }
        }
        if (!answered) sinResponder++;
    });

    const total = data.quiz.length;
    const score = Math.round((correctas / total) * 100);

    // Read PPM saved earlier in this same session (Fase 2)
    const blockNum = window.location.pathname.match(/Bloque(\d+)/) ? window.location.pathname.match(/Bloque(\d+)/)[1] : '1';
    const ppmKey = 'mm_b' + blockNum + '_d' + dayId + '_ppm';
    const ppm = localStorage.getItem(ppmKey) || '—';

    // Save score + mark day as completed (alimenta Mi Kit)
    localStorage.setItem('mm_b' + blockNum + '_d' + dayId + '_score', score.toString());
    localStorage.setItem('mm_b' + blockNum + '_d' + dayId, '1');

    const box = document.getElementById('quiz-result-' + dayId);
    if (!box) return;
    box.style.display = 'block';

    let nivel = "";
    if (score >= 80) nivel = "Comprensión sólida";
    else if (score >= 60) nivel = "Comprensión parcial — repasa lo que falló";
    else nivel = "Fuga de comprensión — vuelve a leer antes de avanzar";

    const color = score >= 80 ? 'var(--success)' : (score >= 60 ? 'var(--accent-fire)' : '#FF4500');
    box.style.borderColor = color;
    box.style.background = 'rgba(255,255,255,0.03)';

    let detalle = sinResponder > 0
        ? `<p style="font-size:0.78rem; color: var(--text-tertiary); margin-top:6px;">${sinResponder} pregunta(s) sin responder — cuentan como incorrectas.</p>`
        : '';

    box.innerHTML = `
        <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:10px;">
            <div>
                <div style="font-family:'Outfit'; font-size:1.8rem; color:${color}; font-weight:700;">${score}%</div>
                <div style="font-size:0.75rem; color:var(--text-secondary); text-transform:uppercase; letter-spacing:1px;">Comprensión (${correctas}/${total})</div>
            </div>
            <div style="text-align:right;">
                <div style="font-family:'Outfit'; font-size:1.8rem; color:var(--accent-water); font-weight:700;">${ppm}</div>
                <div style="font-size:0.75rem; color:var(--text-secondary); text-transform:uppercase; letter-spacing:1px;">PPM leídas</div>
            </div>
        </div>
        <p style="font-size:0.85rem; color:white; font-weight:500;">${nivel}</p>
        ${detalle}
    `;
    box.scrollIntoView({ behavior: 'smooth', block: 'center' });

    // Refresh Mi Kit stats if that view exists
    if (typeof window.renderKitMentoria === 'function') window.renderKitMentoria();
}
